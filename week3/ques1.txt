#include <iostream>
using namespace std;

void insertionSort(int arr[], int n, int& comparisons, int& shifts) {
    comparisons = 0;
    shifts = 0;
    for (int i = 1; i < n; ++i) {
        int key = arr[i];
        int j = i - 1;
        while (j >= 0 && key < arr[j]) {
            comparisons++;
            arr[j + 1] = arr[j];
            j--;
            shifts++;
        }
        if (j >= 0) {
            comparisons++;
        }
        arr[j + 1] = key;
    }
}

int main() {
    int T;
    cout << "Enter number of test cases: ";
    cin >> T;
    for (int t = 0; t < T; ++t) {
        int n;
        cout << "Enter size of array for test case " << t + 1 << ": ";
        cin >> n;
        int arr[n];
        cout << "Enter space-separated integers for the array: ";
        for (int i = 0; i < n; ++i) {
            cin >> arr[i];
        }
        int comparisons, shifts;
        insertionSort(arr, n, comparisons, shifts);
        cout << "Sorted array: ";
        for (int i = 0; i < n; ++i) {
            cout << arr[i] << " ";
        }
        cout << endl;
        cout << "Total comparisons: " << comparisons << endl;
        cout << "Total shifts: " << shifts << endl;
    }
    return 0;
}
