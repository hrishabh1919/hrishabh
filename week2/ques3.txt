#include <iostream>
#include <algorithm>
#include <cstdlib> // For abs() function
using namespace std;

int countPairs(int arr[], int n, int key) {
    int count = 0;

    for (int i = 0; i < n; ++i) {
        int diff = arr[i] - key;
        int upper = upper_bound(arr + i + 1, arr + n, abs(diff)) - arr;
        int lower = lower_bound(arr + i + 1, arr + n, abs(diff)) - arr;
        count += (upper - lower);
    }

    return count;
}

int main() {
    int Tno;
    cout << "Enter the no of test case ";
    cin >> Tno;

    while (Tno--) {
        int n, key;
        cout << "Enter the size of the array ";
        cin >> n;
        int arr[n];

        cout << "Enter the sorted elements of array ";
        for (int i = 0; i < n; ++i)
            cin >> arr[i];

        cout << "Enter the key: ";
        cin >> key;
        cout << "No of pairs with difference equal to " << key << ": " << countPairs(arr, n, key) << endl;
    }

    return 0;
}
