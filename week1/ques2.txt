#include <iostream>

using namespace std;

int binarySearch(int arr[], int n, int target) {
    int low = 0, high = n - 1;

    while (low <= high) {
        int mid = low + (high - low) / 2;

        if (arr[mid] == target)
            return mid;
        else if (arr[mid] < target)
            low = mid + 1;
        else
            high = mid - 1;
    }

    return -1; // Element not found
}

int main() {
    const int maxSize = 100; // Adjust the maximum size as needed
    int t, n, target;
    int arr[maxSize];

    cout << "Enter the number of test cases: ";
    cin >> t;

    for (int i = 0; i < t; ++i) {
        cout << "Enter the size of array for test case " << i + 1 << ": ";
        cin >> n;

        cout << "Enter the sorted array for test case " << i + 1 << ": ";
        for (int j = 0; j < n; ++j)
            cin >> arr[j];

        cout << "Enter the target element for test case " << i + 1 << ": ";
        cin >> target;

        int result = binarySearch(arr, n, target);
        if (result != -1)
            cout << "Element " << target << " found at index " << result << " in test case " << i + 1 << endl;
        else
            cout << "Element " << target << " not found in test case " << i + 1 << endl;
    }

    return 0;
}
